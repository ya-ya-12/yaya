### a005. Eva 的回家作業
https://zerojudge.tw/ShowProblem?problemid=a005
```CPP
#include <iostream>
using namespace std;

int main() {
    int t;
    cin >> t; 

    for (int i = 0; i < t; i++) {
        int a, b, c, d; 
        
        cin >> a >> b >> c >> d;
        if (b - a == c - b && c - b == d - c) {
        
            int e = d + (d - c);
            cout << a << " " << b << " " << c << " " << d << " " << e <<'\n';
        }
        else if (b / a == c / b && c / b == d / c) {
        
            int e = d * (d / c);
            cout << a << " " << b << " " << c << " " << d << " " << e <<'\n';
        }
    }

    return 0;
}
```
